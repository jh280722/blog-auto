1. 확장 프로그램 연결

권장 순서: PREPARE_EDITOR → WRITE_POST
`captcha_required`가 오면 DIRECT STATE / CAPTCHA ARTIFACTS로 막힌 탭과 캡차 이미지를 가져온 뒤, OCR/비전으로 답안을 구하고 `SUBMIT_CAPTCHA_AND_RESUME`를 우선 호출하세요. 답안 입력과 같은 탭 재개를 한 번에 처리합니다. 구버전 호환이 필요할 때만 `SUBMIT_CAPTCHA` → `RESUME_DIRECT_PUBLISH`를 따로 사용하세요.

4. 프로그래밍 연동 가이드

JavaScript에서 아래 코드로 확장 프로그램에 데이터를 보낼 수 있습니다.

const EXTENSION_ID = "YOUR_EXTENSION_ID"; // 1) 에디터 준비 chrome.runtime.sendMessage(EXTENSION_ID, { action: "PREPARE_EDITOR", data: { blogName: "your-blog-name" } }, (response) => console.log(response)); // 2) 단건 발행 chrome.runtime.sendMessage(EXTENSION_ID, { action: "WRITE_POST", data: { blogName: "your-blog-name", title: "글 제목", content: "<p>본문 HTML</p>", category: "카테고리명", tags: ["태그1", "태그2"], visibility: "public", images: [{url: "https://..."}], autoPublish: true } }, (response) => { console.log(response); // captcha_required면 blocked tab/state 확인 if (response.status === "captcha_required") { chrome.runtime.sendMessage(EXTENSION_ID, { action: "GET_DIRECT_PUBLISH_STATE", data: { includeCaptchaContext: true } }, (state) => console.log(state)); } }); // CAPTCHA 컨텍스트 직접 조회 chrome.runtime.sendMessage(EXTENSION_ID, { action: "GET_CAPTCHA_CONTEXT" }, (response) => console.log(response)); // CAPTCHA 이미지 아티팩트 조회 (saved blocked tab 우선) chrome.runtime.sendMessage(EXTENSION_ID, { action: "GET_CAPTCHA_ARTIFACTS" }, (response) => console.log(response.artifact)); // CAPTCHA 답안 제출 + 같은 탭 재개 (권장) chrome.runtime.sendMessage(EXTENSION_ID, { action: "SUBMIT_CAPTCHA_AND_RESUME", data: { answer: "1234" } }, (response) => { console.log(response.submitResult, response.resumeResult || response); }); // 구버전 호환: CAPTCHA 답안만 제출 chrome.runtime.sendMessage(EXTENSION_ID, { action: "SUBMIT_CAPTCHA", data: { answer: "1234" } }, (response) => { console.log(response); if (!response.captchaStillAppears) { chrome.runtime.sendMessage(EXTENSION_ID, { action: "RESUME_DIRECT_PUBLISH" }, (resume) => console.log(resume)); } }); // 큐에 추가 chrome.runtime.sendMessage(EXTENSION_ID, { action: "ADD_TO_QUEUE", data: [{ title: "...", ... }] }, (response) => console.log(response)); // 큐 처리 시작 chrome.runtime.sendMessage(EXTENSION_ID, { action: "START_QUEUE" }, (response) => console.log(response));
